    def match_capitalized(phrase_1: str, phrase_2: str) -> str:
        """
        Matches capitalized words in the translated phrase to the original phrase.

        :param phrase_1: The original phrase.
        :type phrase_1: str
        :param phrase_2: The translated phrase.
        :type phrase_2: str
        :return: The translated phrase with capitalized words matched.
        :rtype: str
        """
        set_all_upper = set()  # Set to hold words that should be capitalized
        for word in phrase_1.split():
            upper_counter = sum(1 for char in word if char.isupper())  # Count uppercase letters in the word

            if (upper_counter > 2 and len(word) > 2) or (upper_counter > 1 and len(word) < 3):
                set_all_upper.add(word)  # Add the word to the set if it meets the conditions

        words = phrase_2.split()
        for i, word in enumerate(words):
            if word.upper() in set_all_upper:  # Check if the word should be capitalized
                words[i] = word.upper()  # Capitalize the word
        phrase_2 = ' '.join(words)
        return phrase_2



[docs]    def traduction_report(self, output_path: str, save: bool) -> Document:
        """
        Translates the content of the document and saves it to the specified output path.

        :param output_path: The file path where the translated document will be saved.
        :type output_path: str
        :param save: True if user wishes to save translated document to output_path.
        :type save: bool
        :return: The translated document.
        :rtype: Document
        """
        paragraphs = [paragraph for paragraph in self.doc.paragraphs if
                      len(paragraph.text) > 0]  # Get non-empty paragraphs
        for paragraph in tqdm(paragraphs):  # Iterate over each paragraph
            runs = paragraph.runs
            for run in runs:  # Iterate over each run
                if len(run.text) > 2 and run.text != ' ':  # Check if the run has meaningful text
                    old_text = run.text
                    new_text = self.traduction_google(old_text)  # Translate the text
                    if new_text is None:  # If translation fails, skip
                        continue
                    new_text = self.match_capitalized(old_text, new_text)  # Match capitalization
                    run.text = run.text.replace(old_text, new_text)  # Replace the old text with the new text

        for table in tqdm(self.doc.tables):  # Iterate over each table in the document
            for row in table.rows:
                for cell in row.cells:
                    paragraphs = [paragraph for paragraph in cell.paragraphs if
                                  len(paragraph.text) > 0]  # Get non-empty paragraphs in cell
                    for paragraph in paragraphs:  # Iterate over each paragraph in the cell
                        runs = paragraph.runs
                        for run in runs:  # Iterate over each run
                            if len(run.text) > 2 and run.text != ' ':  # Check if the run has meaningful text
                                old_text = run.text  # Store the original text
                                new_text = self.traduction_google(old_text)  # Translate the text
                                if new_text is None:  # If translation fails, skip
                                    continue
                                new_text = self.match_capitalized(old_text, new_text)  # Match capitalization
                                run.text = run.text.replace(old_text,
                                                            new_text)  # Replace the old text with the new text

        if save:
            self.doc.save(output_path)  # Save the translated document
        return self.doc

[docs]    def save_images(self) -> list[str]:
        """
        Parses an Excel file and saves all of its images in a separate folder.

        :return: A list of references for saved images.
        :rtype: list[str]
        """
        in_mem_file = None
        with open(self.xlsx_file_path, "rb") as f:
            in_mem_file = io.BytesIO(f.read())
        pxl_doc = openpyxl.load_workbook(in_mem_file)
        excel_filename = os.path.basename(self.xlsx_file_path)
        os.makedirs(f'data\\images\\{excel_filename}', exist_ok=True)
        sheet = pxl_doc[pxl_doc.sheetnames[0]]
        image_loader = SheetImageLoader(sheet)
        cell_dic = {i: f'I{i}' for i in range(sheet.max_row + 1)}
        ref_list = []
        for row, cell in cell_dic.items():
            if image_loader.image_in(cell):
                image = image_loader.get(cell)
                ref = sheet[f'A{row}'].value
                ref_list.append(ref)
                image.save(f'data\\images\\{excel_filename}\\{ref}.png')
        return ref_list



[docs]    def format_df_report(self) -> pd.DataFrame:
        """
        Reads an Excel file, formats it, and returns a pandas DataFrame.

        :return: The formatted DataFrame.
        :rtype: pd.DataFrame
        """
        df = pd.read_excel(self.xlsx_file_path, index_col=0, usecols=[i for i in range(9)])
        ref_list = self.save_images()
        excel_filename = os.path.basename(self.xlsx_file_path)
        for ref in ref_list:
            df.loc[ref, 'Illustration '] = f'data\\images\\{excel_filename}\\{ref}.png'
        formatted_df = df.drop(columns=['Valeur', 'Criticité', 'Catégorie ']).dropna(how='all', axis=1)
        return formatted_df.reset_index()

    def format_doc_report(table) -> None:
        """
        Formats the table (font, alignments, cell background color, etc.).

        :param table: The table to format.
        """
        for row_index, row in enumerate(table.rows):
            for cell_index, cell in enumerate(row.cells):
                for paragraph in cell.paragraphs:
                    if cell_index == 2 or row_index == 0:
                        paragraph.alignment = WD_ALIGN_PARAGRAPH.CENTER
                    else:
                        paragraph.alignment = WD_ALIGN_PARAGRAPH.LEFT
                    for run in paragraph.runs:
                        run.font.size = Pt(10)
                        if row_index == 0:
                            run.bold = True
                            run.font.name = 'Hellix'
                            color = "#008AC9"
                            cell._element.get_or_add_tcPr().append(parse_xml(
                                r'<w:shd {} w:fill="{}"/>'.format(nsdecls('w'), color)))
                        else:
                            run.bold = False
                            run.font.name = 'Hellix-Regular'

[docs]    def generate_table(self, doc, output_path: str, translation: str, save: bool = True) -> None:
        """
        Generates a table from the formatted DataFrame and saves it to a Word document.

        :param doc: The Word document to which the table will be added.
        :param output_path: The output path for the generated document.
        :type output_path: str
        :param translation: Translation language. Set to 'no' for no translation.
        :type translation: str
        :param save: True if the document should be saved, False otherwise.
        :type save: bool
        """
        df = self.format_df_report()
        table = doc.add_table(rows=1, cols=df.shape[1], style='Table Grid')
        header = table.rows[0].cells
        for j, column in enumerate(df.columns):
            header[j].text = column
        for i, cell in enumerate(header):
            if i == 5:
                cell.vertical_alignment = WD_ALIGN_VERTICAL.CENTER
            if i == 4:
                for paragraph in cell.paragraphs:
                    paragraph.alignment = WD_ALIGN_PARAGRAPH.LEFT
            else:
                for paragraph in cell.paragraphs:
                    paragraph.alignment = WD_ALIGN_PARAGRAPH.CENTER
        for _, row in df.iterrows():
            cells = table.add_row().cells
            for cell, value in zip(cells, row):
                if str(value).endswith('.png'):
                    paragraph = cell.paragraphs[0]
                    paragraph.alignment = WD_ALIGN_PARAGRAPH.CENTER
                    run = paragraph.add_run()
                    run.add_picture(value, width=Inches(1.8))
                    cell.vertical_alignment = WD_ALIGN_VERTICAL.CENTER
                elif str(value) == 'nan':
                    cell.text = ''
                else:
                    cell.text = str(value)
        self.adjust_col_width(table, 5, Inches(3))
        self.adjust_col_width(table, 1, Inches(2))
        self.adjust_col_width(table, 3, Inches(3))
        self.set_repeat_table_header(table.rows[0])
        self.format_doc_report(table)
        self.prevent_document_break(doc)
        paragraph = self.find_paragraph_precede_rectable(doc)
        self.move_table_after(table, paragraph)
        if translation != 'no':
            doc_translation = DocumentTranslation(doc, 'fr', translation)
            doc_translation.traduction_report(output_path, save)
        if save:
            doc.save(output_path)
        else:
            return table

    def copy_cell_properties(source_cell, dest_cell) -> None:
        """
        Copies cell properties from a source cell to a destination cell.

        Copies cell background shading, borders, etc., in a python-docx Document.

        :param source_cell: The source cell with desired formatting.
        :type source_cell: docx.table._Cell
        :param dest_cell: The destination cell to which to apply formatting.
        :type dest_cell: docx.table._Cell
        """
        # Get properties from source cell
        # (tcPr = table cell properties)
        cell_properties = source_cell._tc.get_or_add_tcPr()

        # Remove any table cell properties from destination cell
        # (otherwise, we end up with two <w:tcPr> ... </w:tcPr> blocks)
        dest_cell._tc.remove(dest_cell._tc.get_or_add_tcPr())

        # Make a shallow copy of the source cell properties
        # (otherwise, properties get *moved* from source to destination)
        cell_properties = copy.copy(cell_properties)

        # Append the copy of properties from the source cell to the destination cell
        dest_cell._tc.append(cell_properties)


    @property
    def table_length(self):
        """
        Returns the table length of the visitor table.

        :return: The length of the visitor table.
        :rtype: int
        """
        df = pd.read_excel(self.replace_dic_path)
        table_length = df.loc[df['Code du champ'] == 'nom_visiteur'].iat[0, 4]
        return table_length


[docs]    def change_people_table(self, table) -> None:
        """
        Modifies the provided table by duplicating rows and updating cell content based on specific conditions.

        :param table: A table object that supports adding rows and accessing cells through rows.
        :type table: Table
        """
        l = self.table_length
        if l > 1:
            for i in range(2, l + 1):
                table.add_row()
                new_row = table.rows[i]
                old_row = table.rows[i - 1]
                for old_cell, new_cell in zip(old_row.cells, new_row.cells):
                    self.copy_cell_properties(old_cell, new_cell)
                    if 'ate' in (old_cell.text) or 'ITE' in (old_cell.text):
                        new_cell.text = old_cell.text
                    else:
                        new_cell.text = old_cell.text[:-2] + f'{int(old_cell.text[-2]) + 1}' + '}'

[docs]    def mastery_level(self):
        """
        Determines the mastery level based on the frequency and gravity values.

        :return: The mastery level as a string.
        :rtype: str
        """
        freq, grav = self.freq_grav
        if freq == grav == 4:
            return "Maîtrise médiocre"
        elif 2 <= freq <= 3 and grav == 3:
            return "Maîtrise faible"
        elif 2 <= freq <= 3 and grav == 2:
            return "Maîtrise moyenne"
        else:
            return "Maîtrisé"
